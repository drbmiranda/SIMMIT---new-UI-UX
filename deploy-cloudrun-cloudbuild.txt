PROJECT_ID="SEU_PROJECT_ID"
REGION="us-central1"
REPO_OWNER="drbmiranda"
REPO_NAME="SIMMIT---new-UI-UX"
BRANCH="^main$"
SERVICE="medsim-beta"
AR_REPO="simmit-images"
TRIGGER_NAME="${SERVICE}-main"

SECRET_GEMINI="simmit-gemini-api-key"
SECRET_SUPABASE_URL="simmit-supabase-url"
SECRET_SUPABASE_ANON="simmit-supabase-anon-key"

gcloud config set project "${PROJECT_ID}"

# 1) APIs
gcloud services enable \
  cloudbuild.googleapis.com \
  run.googleapis.com \
  artifactregistry.googleapis.com \
  secretmanager.googleapis.com

# 2) Artifact Registry (uma vez)
gcloud artifacts repositories create "${AR_REPO}" \
  --repository-format=docker \
  --location="${REGION}" \
  --description="Docker images for ${SERVICE}" || true

# 3) IAM para Cloud Build SA
PROJECT_NUMBER="$(gcloud projects describe "${PROJECT_ID}" --format='value(projectNumber)')"
CLOUDBUILD_SA="${PROJECT_NUMBER}@cloudbuild.gserviceaccount.com"

gcloud projects add-iam-policy-binding "${PROJECT_ID}" --member="serviceAccount:${CLOUDBUILD_SA}" --role="roles/run.admin"
gcloud projects add-iam-policy-binding "${PROJECT_ID}" --member="serviceAccount:${CLOUDBUILD_SA}" --role="roles/artifactregistry.writer"
gcloud projects add-iam-policy-binding "${PROJECT_ID}" --member="serviceAccount:${CLOUDBUILD_SA}" --role="roles/secretmanager.secretAccessor"
gcloud projects add-iam-policy-binding "${PROJECT_ID}" --member="serviceAccount:${CLOUDBUILD_SA}" --role="roles/iam.serviceAccountUser"

# 4) Secrets (crie com valor real)
printf 'SUA_GEMINI_KEY' | gcloud secrets create "${SECRET_GEMINI}" --data-file=- || printf 'SUA_GEMINI_KEY' | gcloud secrets versions add "${SECRET_GEMINI}" --data-file=-
printf 'SUA_SUPABASE_URL' | gcloud secrets create "${SECRET_SUPABASE_URL}" --data-file=- || printf 'SUA_SUPABASE_URL' | gcloud secrets versions add "${SECRET_SUPABASE_URL}" --data-file=-
printf 'SUA_SUPABASE_ANON_KEY' | gcloud secrets create "${SECRET_SUPABASE_ANON}" --data-file=- || printf 'SUA_SUPABASE_ANON_KEY' | gcloud secrets versions add "${SECRET_SUPABASE_ANON}" --data-file=-

# 5) Trigger GitHub (branch main)
gcloud builds triggers create github \
  --name="${TRIGGER_NAME}" \
  --repo-owner="${REPO_OWNER}" \
  --repo-name="${REPO_NAME}" \
  --branch-pattern="${BRANCH}" \
  --build-config="cloudbuild.yaml" \
  --substitutions="_SERVICE=${SERVICE},_REGION=${REGION},_AR_REPO=${AR_REPO},_SECRET_GEMINI_API_KEY=${SECRET_GEMINI},_SECRET_SUPABASE_URL=${SECRET_SUPABASE_URL},_SECRET_SUPABASE_ANON_KEY=${SECRET_SUPABASE_ANON}"
